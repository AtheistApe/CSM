\section*{Some possibly useful identities}
\emph{You will \textbf{need} to know these and other identities for the exam}
\begin{itemize}
  \item $1 + \cot^2\theta = \csc^2\theta$
  \item $1 + \tan^2\theta = \sec^2\theta$
  \item $\ds\int\sec\theta\;d\theta = \ln|\sec\theta + \tan\theta| + C$
  \item $\ds\int\csc\theta\;d\theta = -\ln|\csc\theta + \cot\theta| + C$
  \item $\ds\int\tan\theta\;d\theta = \ln|\sec\theta| + C$
  \item $\ds\int\cot\theta\;d\theta = -\ln|\csc\theta| + C$
\end{itemize}
